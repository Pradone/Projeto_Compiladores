GRAMÁTICA AJUSTADA DA LINGUAGEM

Prog -> PROGRAM IDENTIFIER PVIG Decls CmdComp PONTO
Decls -> VAR ListDecl
       | ε
ListDecl -> DeclTip ListDecl
          | DeclTip
DeclTip -> ListId DPONTOS Tip PVIG
ListId -> IDENTIFIER
        | IDENTIFIER VIG ListId
Tip -> INTEGER
     | BOOLEAN
CmdComp -> BEGIN ListCmd END
ListCmd -> Cmd
         | Cmd PVIG ListCmd
Cmd -> CmdIf
     | CmdWhile
     | CmdRead
     | CmdWrite
     | CmdAtrib
     | CmdComp
CmdIf -> IF ExprLogica THEN Cmd ElsePart
ElsePart -> ELSE Cmd
          | ε
CmdWhile -> WHILE ExprLogica DO Cmd
CmdRead -> READ ABPAR ListId FPAR
CmdWrite -> WRITE ABPAR ListW FPAR
ListW -> ElemW
       | ElemW VIG ListW
ElemW -> ExprLogica
       | CADEIA
CmdAtrib -> IDENTIFIER ATRIB ExprLogica
ExprLogica -> ExprRel ExprLogica2
ExprLogica2 -> OPLOG ExprRel ExprLogica2
             | ε
ExprRel -> ExprArit ExprRel2
ExprRel2 -> OPREL ExprArit
          | ε
ExprArit -> Termo ExprArit2
ExprArit2 -> OPAD Termo ExprArit2
           | ε
Termo -> Fator Termo2
Termo2 -> OPMULT Fator Termo2
        | ε
Fator -> IDENTIFIER
       | CTE
       | TRUE
       | FALSE
       | OPNEG Fator
       | ABPAR ExprLogica FPAR
